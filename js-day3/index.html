JavaScript Arrays & Objects - Assignments

Arrays - Common Methods & Examples

  • push() → Adds element at end
    let arr = [1,2];
arr.push(3);
console.log(arr); // [1,2,3]
  • pop() → Removes last element
    let arr = [1,2,3];
arr.pop();
console.log(arr); // [1,2]
  • shift() → Removes first element
    let arr = [1,2,3];
arr.shift();
console.log(arr); // [2,3]
  • unshift() → Adds element at start
    let arr = [2,3];
arr.unshift(1);
console.log(arr); // [1,2,3]
  • concat() → Combine arrays
    let arr1 = [1,2];
let arr2 = [3,4];
let arr3 = arr1.concat(arr2);
console.log(arr3); // [1,2,3,4]
  • slice() → Extract part of array
    let arr = [1,2,3,4];
console.log(arr.slice(1,3)); // [2,3]
  • splice() → Remove or replace elements
    let arr = [1,2,3,4];
arr.splice(1,2,8,9);
console.log(arr); // [1,8,9,4]
  • indexOf() → Find index
    let arr = [1,2,3];
console.log(arr.indexOf(2)); // 1
  • includes() → Check existence
    let arr = [1,2,3];
console.log(arr.includes(2)); // true
  • forEach() → Loop through array
    let arr = [1,2,3];
arr.forEach(el => console.log(el));
// 1
// 2
// 3

Objects - Common Methods & Examples

  • Object.keys() → Get all keys
    let obj = {a:1, b:2};
console.log(Object.keys(obj)); // ["a","b"]
  • Object.values() → Get all values
    console.log(Object.values(obj)); // [1,2]
  • Object.entries() → Get key-value pairs
    console.log(Object.entries(obj)); // [["a",1],["b",2]]
  • delete → Remove property
    delete obj.a;
console.log(obj); // {b:2}
  • hasOwnProperty() → Check property
    console.log(obj.hasOwnProperty("b")); // true
  • for...in → Loop through keys
    for(let key in obj){
  console.log(key, obj[key]);
}
// b 2
  • Object.assign() → Merge objects
    let obj2 = {c:3};
let merged = Object.assign(obj, obj2);
console.log(merged); // {b:2, c:3}
  • JSON.stringify() → Convert object to string
    console.log(JSON.stringify(obj)); // '{"b":2,"c":3}'
  • JSON.parse() → Convert string to object
    let str = '{"x":10}';
let parsed = JSON.parse(str);
console.log(parsed); // {x:10}
  • Object.freeze() → Prevent modification
    Object.freeze(obj);
obj.b = 100; // ignored
console.log(obj); // {b:2,c:3}
