Objects - Common Methods & Examples

  • Object.keys() → Get all keys
    let obj = {a:1, b:2};
console.log(Object.keys(obj)); // ["a","b"]
  • Object.values() → Get all values
    console.log(Object.values(obj)); // [1,2]
  • Object.entries() → Get key-value pairs
    console.log(Object.entries(obj)); // [["a",1],["b",2]]
  • delete → Remove property
    delete obj.a;
console.log(obj); // {b:2}
  • hasOwnProperty() → Check property
    console.log(obj.hasOwnProperty("b")); // true
  • for...in → Loop through keys
    for(let key in obj){
  console.log(key, obj[key]);
}
// b 2
  • Object.assign() → Merge objects
    let obj2 = {c:3};
let merged = Object.assign(obj, obj2);
console.log(merged); // {b:2, c:3}
  • JSON.stringify() → Convert object to string
    console.log(JSON.stringify(obj)); // '{"b":2,"c":3}'
  • JSON.parse() → Convert string to object
    let str = '{"x":10}';
let parsed = JSON.parse(str);
console.log(parsed); // {x:10}
  • Object.freeze() → Prevent modification
    Object.freeze(obj);
obj.b = 100; // ignored
console.log(obj); // {b:2,c:3}
