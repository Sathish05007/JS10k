// toFixed(): rounds a number to fixed decimals

console.log("toFixed():", num.toFixed());
console.log("toFixed(10):", num.toFixed(10));

console.log(parseInt("45.67"));

// Number.isFinite(): checks if value is finite number
console.log(Number.isFinite(10 / 2));

// Global Scope
var a = 20;
function sum() {
  console.log("Inside function:", a);
  if (true) {
    console.log("Inside block:", a);
  }
}
sum();
console.log("Outside function:", a);

// Block Scope
if (true) {
  let x = 22;
  console.log("Inside block:", x);
}
// console.log(x); // Error if uncommented

// Lexical Scope Example
function add(a, b, c) {
  if (a === 13) return 0;
  else if (b === 13) return a;
  else if (a === 13 || b === 13 || c === 13) return "can't do anything";
  else return a + b + c;
}
console.log("All below 13:", add(1, 2, 3));
console.log("a=13:", add(13, 2, 3));
console.log("b=13:", add(1, 13, 3));
console.log("c=13:", add(1, 2, 13));
